001package edu.hawaii.jmotif.sax;
002
003import java.util.Arrays;
004import java.util.Set;
005import org.slf4j.LoggerFactory;
006import ch.qos.logback.classic.Level;
007import ch.qos.logback.classic.Logger;
008import edu.hawaii.jmotif.sax.alphabet.NormalAlphabet;
009import edu.hawaii.jmotif.sax.datastructures.SAXRecords;
010import edu.hawaii.jmotif.sax.parallel.ParallelSAXImplementation;
011import edu.hawaii.jmotif.util.StackTrace;
012
013/**
014 * This implements a simple CLI tool for ad-hoc SAX discretization.
015 * 
016 * @author Pavel Senin
017 * 
018 */
019public class SAXcliConverter {
020
021  private static final String CR = "\n";
022  private static final String COMMA = ", ";
023
024  // logging stuff
025  //
026  private static Logger consoleLogger;
027  private static Level LOGGING_LEVEL = Level.INFO;
028  static {
029    consoleLogger = (Logger) LoggerFactory.getLogger(SAXcliConverter.class);
030    consoleLogger.setLevel(LOGGING_LEVEL);
031  }
032
033  /**
034   * The main runnable.
035   * 
036   * @param args
037   */
038  public static void main(String[] args) {
039
040    try {
041
042      if (args.length < 6 || args.length > 7) {
043        System.out.println(printHelp());
044        System.exit(-1);
045      }
046
047      String dataFName = args[0];
048      double[] ts = TSProcessor.readFileColumn(dataFName, 0, 0);
049
050      Integer slidingWindowSize = Integer.valueOf(args[1]);
051      Integer paaSize = Integer.valueOf(args[2]);
052      Integer alphabetSize = Integer.valueOf(args[3]);
053
054      NumerosityReductionStrategy nrStrategy = NumerosityReductionStrategy.valueOf(args[4]
055          .toUpperCase());
056
057      Double nThreshold = Double.valueOf(args[5]);
058
059      NormalAlphabet na = new NormalAlphabet();
060      SAXProcessor sp = new SAXProcessor();
061
062      Integer threadsNum = 1;
063      SAXRecords res = null;
064
065      if (args.length > 6) {
066        threadsNum = Integer.valueOf(args[6]);
067      }
068
069      if (threadsNum > 1) {
070        ParallelSAXImplementation ps = new ParallelSAXImplementation();
071        res = ps.process(ts, threadsNum, slidingWindowSize, paaSize, alphabetSize, nrStrategy,
072            nThreshold);
073      }
074      else {
075        res = sp.ts2saxViaWindow(ts, slidingWindowSize, paaSize, na.getCuts(alphabetSize),
076            nrStrategy, nThreshold);
077      }
078
079      Set<Integer> index = res.getIndexes();
080      for (Integer idx : index) {
081        System.out.println(idx + COMMA + String.valueOf(res.getByIndex(idx).getPayload()));
082      }
083
084    }
085    catch (Exception e) {
086      System.err.println("error occured while parsing parameters " + Arrays.toString(args) + CR
087          + StackTrace.toString(e));
088      System.exit(-1);
089    }
090
091  }
092
093  /**
094   * Prints the CLI help.
095   * 
096   * @return formatted help string.
097   */
098  private static String printHelp() {
099    StringBuffer sb = new StringBuffer();
100    sb.append("Command-line SAX conversion utility, the output printed to STDOUT ").append(CR);
101    sb.append("Expects 6 parameters:").append(CR);
102    sb.append(" [1] training dataset filename").append(CR);
103    sb.append(" [2] sliding window size").append(CR);
104    sb.append(" [3] PAA size").append(CR);
105    sb.append(" [4] Alphabet size").append(CR);
106    sb.append(" [5] numerosity reduction <NONE|EXACT|MINDIST>").append(CR);
107    sb.append(" [6] z-Normalization threshold value").append(CR);
108    sb.append(" [7] OPTIONAL: number of threads to use").append(CR);
109    sb.append("An execution example: $java -jar \"jmotif-vsm-20.jar\" ");
110    sb.append("test/data/ecg0606_1.csv 120 7 5 EXACT 0.001 2").append(CR);
111    return sb.toString();
112  }
