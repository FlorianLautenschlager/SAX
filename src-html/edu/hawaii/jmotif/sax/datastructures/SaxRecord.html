001package edu.hawaii.jmotif.sax.datastructures;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005
006/**
007 * Implement a data container for the parallel SAX.
008 * 
009 * @author psenin
010 * 
011 */
012public class SaxRecord implements Comparable<SaxRecord> {
013
014  /** The payload. */
015  private char[] saxString;
016
017  /** The index of occurrences in the raw sequence. */
018  private ArrayList<Integer> occurrences;
019
020  /** Disable the constructor. */
021  @SuppressWarnings("unused")
022  private SaxRecord() {
023    super();
024  }
025
026  /**
027   * The allowed constructor.
028   * 
029   * @param str the payload value.
030   * @param idx the occurrence index.
031   */
032  public SaxRecord(char[] str, int idx) {
033    super();
034    this.saxString = str.clone();
035    this.occurrences = new ArrayList<Integer>();
036    this.addIndex(idx);
037  }
038
039  /**
040   * Adds an index.
041   * 
042   * @param idx The index to add.
043   */
044  public void addIndex(int idx) {
045    if (!(this.occurrences.contains(idx))) {
046      this.occurrences.add(idx);
047    }
048  }
049
050  /**
051   * Removes a single index.
052   * 
053   * @param idx The index to remove.
054   */
055  public void removeIndex(Integer idx) {
056    this.occurrences.remove(idx);
057  }
058
059  /**
060   * Gets the payload of the structure.
061   * 
062   * @return The string.
063   */
064  public char[] getPayload() {
065    return this.saxString;
066  }
067
068  /**
069   * Get all indexes.
070   * 
071   * @return all indexes.
072   */
073  public ArrayList<Integer> getIndexes() {
074    return this.occurrences;
075  }
076
077  /**
078   * This comparator compares entries by the length of the entries array - i.e. by the total
079   * frequency of entry occurrence.
080   * 
081   * @param o an entry to compare with.
082   * @return results of comparison.
083   */
084  @Override
085  public int compareTo(SaxRecord o) {
086    int a = this.occurrences.size();
087    int b = o.getIndexes().size();
088    if (a == b) {
089      return 0;
090    }
091    else if (a > b) {
092      return 1;
093    }
094    return -1;
095  }
096
097  /**
098   * {@inheritDoc}
099   */
100  @Override
101  public boolean equals(Object o) {
102    if (o instanceof SaxRecord) {
103      SaxRecord other = (SaxRecord) o;
104      if (Arrays.equals(other.getPayload(), this.saxString)
105          && (other.getIndexes().size() == this.occurrences.size())) {
106        for (Integer e : this.occurrences) {
107          if (!other.getIndexes().contains(e)) {
108            return false;
109          }
110        }
111        return true;
112      }
113    }
114    return false;
115  }
116
117  /**
118   * {@inheritDoc}
119   */
120  @Override
121  public int hashCode() {
122    int hash = 7;
123    int num0 = 0;
124    if (null == this.saxString || 0 == this.saxString.length) {
125      num0 = 32;
126    }
127    else {
128      for (int i = 0; i < this.saxString.length; i++) {
129        num0 = num0 + Character.getNumericValue(this.saxString[i]);
130      }
131    }
132
133    int num1 = 0;
134    if (this.occurrences.isEmpty()) {
135      num1 = 17;
136    }
137    else {
138      for (Integer i : this.occurrences) {
139        num1 = num1 + i;
140      }
141    }
142
143    hash = num0 + hash * num1;
144    return hash;
145  }
146
147  @Override
148  public String toString() {
149    StringBuffer sb = new StringBuffer();
150    sb.append(this.saxString).append(" -> ").append(occurrences.toString());
151    return sb.toString();
152  }
153
154}
