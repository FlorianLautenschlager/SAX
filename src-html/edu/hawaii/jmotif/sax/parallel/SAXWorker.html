001package edu.hawaii.jmotif.sax.parallel;
002
003import java.util.Arrays;
004import java.util.HashMap;
005import java.util.concurrent.Callable;
006import org.slf4j.LoggerFactory;
007import ch.qos.logback.classic.Level;
008import ch.qos.logback.classic.Logger;
009import edu.hawaii.jmotif.sax.NumerosityReductionStrategy;
010import edu.hawaii.jmotif.sax.SAXProcessor;
011import edu.hawaii.jmotif.sax.TSProcessor;
012import edu.hawaii.jmotif.sax.alphabet.NormalAlphabet;
013
014/**
015 * A callable worker class that will translate a given interval of the timeseries into SAX.
016 * 
017 * @author psenin
018 * 
019 */
020public class SAXWorker implements Callable<HashMap<Integer, char[]>> {
021
022  /** The worker ID. */
023  private long id;
024
025  /** The input timeseries. */
026  private double[] data;
027
028  /** The conversion start index. */
029  private int intervalStart;
030
031  /** The conversion end index. */
032  private int intervalEnd;
033
034  /** The SAX discretization sliding window parameter size. */
035  private int saxWindowSize;
036
037  /** The SAX discretization PAA size. */
038  private int saxPAASize;
039
040  /** The SAX discretization alphabet size. */
041  private int saxAlphabetSize;
042
043  /** The SAX discretization sliding window parameter size. */
044  private NumerosityReductionStrategy numerosityReductionStrategy;
045
046  /** The SAX discretization normalization threshold. */
047  private double normalizationThreshold;
048
049  // logging stuff
050  //
051  private static Logger consoleLogger;
052  private static Level LOGGING_LEVEL = Level.INFO;
053
054  // static block - we instantiate the logger
055  //
056  static {
057    consoleLogger = (Logger) LoggerFactory.getLogger(SAXWorker.class);
058    consoleLogger.setLevel(LOGGING_LEVEL);
059  }
060
061  /**
062   * Performs SAX discretization.
063   * 
064   * @param id the job id.
065   * @param data the data array.
066   * @param intervalStart from which coordinate to start the conversion.
067   * @param intervalEnd where to end conversion (non-ionclusive).
068   * @param offset the offset for final values of SAX word positions.
069   * @param windowSize SAX window size.
070   * @param paaSize SAX paa size.
071   * @param alphabetSize SAX alphabet size.
072   * @param nrs The numerosity reduction strategy.
073   * @param normalizationThreshold The normalization strategy.
074   */
075  public SAXWorker(long id, double[] data, int intervalStart, int intervalEnd, int windowSize,
076      int paaSize, int alphabetSize, NumerosityReductionStrategy nrs, double normalizationThreshold) {
077    super();
078    this.id = id;
079    this.data = data;
080    this.intervalStart = intervalStart;
081    this.intervalEnd = intervalEnd;
082    this.saxWindowSize = windowSize;
083    this.saxPAASize = paaSize;
084    this.saxAlphabetSize = alphabetSize;
085    this.numerosityReductionStrategy = nrs;
086    this.normalizationThreshold = normalizationThreshold;
087    consoleLogger.debug("sax worker instance id " + this.id + ", data " + this.data.length
088        + ", window " + this.saxWindowSize + ", paa " + this.saxPAASize + ", alphabet "
089        + this.saxAlphabetSize + ", nr " + this.numerosityReductionStrategy.toString()
090        + ", threshold: " + normalizationThreshold + ", start: " + this.intervalStart + ", end: "
091        + this.intervalEnd);
092  }
093
094  @Override
095  public HashMap<Integer, char[]> call() throws Exception {
096
097    NormalAlphabet na = new NormalAlphabet();
098    TSProcessor tsp = new TSProcessor();
099    SAXProcessor sp = new SAXProcessor();
100
101    HashMap<Integer, char[]> res = new HashMap<Integer, char[]>();
102    res.put(-1, String.valueOf(this.id).toCharArray());
103
104    // scan across the time series extract sub sequences, and convert
105    // them to strings
106    char[] previousString = null;
107    for (int i = this.intervalStart; i < this.intervalEnd - (this.saxWindowSize - 1); i++) {
108
109      // fix the current subsection
110      double[] subSection = Arrays.copyOfRange(this.data, i, i + this.saxWindowSize);
111
112      // Z normalize it
113      subSection = tsp.znorm(subSection, normalizationThreshold);
114
115      // perform PAA conversion if needed
116      double[] paa = tsp.paa(subSection, this.saxPAASize);
117
118      // Convert the PAA to a string.
119      char[] currentString = tsp.ts2String(paa, na.getCuts(this.saxAlphabetSize));
120
121      if (null != previousString) {
122
123        if (NumerosityReductionStrategy.EXACT.equals(this.numerosityReductionStrategy)
124            && Arrays.equals(previousString, currentString)) {
125          // NumerosityReduction
126          continue;
127        }
128        else if (NumerosityReductionStrategy.MINDIST.equals(this.numerosityReductionStrategy)
129            && sp.checkMinDistIsZero(previousString, currentString)) {
130          continue;
131        }
132
133      }
134
135      previousString = currentString;
136
137      res.put(i, currentString);
138
139      consoleLogger.trace(this.id + ", " + String.valueOf(currentString) + ", " + i);
140
141    }
142
143    // System.out.println(this.id + "@" + res.getSAXString(" "));
144    return res;
145
146  }
147}
