003import java.util.Arrays;
004import org.joda.time.Duration;
005import org.joda.time.format.PeriodFormatter;
006import org.joda.time.format.PeriodFormatterBuilder;

009/**
010 * Implements SAX algorithms.
011 * 
012 * @author Pavel Senin
013 * 
014 */
015public final class SAXProcessor {
016
017  private final TSProcessor tsProcessor;
018
019  /**
020   * Constructor.
021   */
022  public SAXProcessor() {
023    super();
024    this.tsProcessor = new TSProcessor();
025  }
026
027  /**
028   * Converts the input time series into a SAX data structure via sliding window and Z
029   * normalization.
030   * 
031   * @param ts the input data.
032   * @param windowSize the sliding window size.
033   * @param paaSize the PAA size.
034   * @param cuts the Alphabet cuts.
035   * @param nThreshold the normalization threshold value.
036   * @param strategy the NR strategy.
037   * 
038   * @return SAX representation of the time series.
039   */
040  public SAXRecords ts2saxViaWindow(double[] ts, int windowSize, int paaSize, double[] cuts,
041      NumerosityReductionStrategy strategy, double nThreshold) {
042
043    // the resulting data structure init
044    //
045    SAXRecords saxFrequencyData = new SAXRecords();
046
047    // scan across the time series extract sub sequences, and convert them to strings
048    char[] previousString = null;
049
050    for (int i = 0; i < ts.length - (windowSize - 1); i++) {
051
052      // fix the current subsection
053      double[] subSection = Arrays.copyOfRange(ts, i, i + windowSize);
054
055      // Z normalize it
056      subSection = tsProcessor.znorm(subSection, nThreshold);
057
058      // perform PAA conversion if needed
059      double[] paa = tsProcessor.paa(subSection, paaSize);
060
061      // Convert the PAA to a string.
062      char[] currentString = tsProcessor.ts2String(paa, cuts);
063
064      if (null != previousString) {
065
066        if (NumerosityReductionStrategy.EXACT.equals(strategy)
067            && Arrays.equals(previousString, currentString)) {
068          // NumerosityReduction
069          continue;
070        }
071        else if (NumerosityReductionStrategy.MINDIST.equals(strategy)
072            && checkMinDistIsZero(previousString, currentString)) {
073          continue;
074        }
075
076      }
077
078      previousString = currentString;
079
080      saxFrequencyData.add(currentString, i);
081    }
082
083    return saxFrequencyData;
084
085  }
086
087  /**
088   * Converts the input time series into a SAX data structure via chunking and Z normalization.
089   * 
090   * @param ts the input data.
091   * @param paaSize the PAA size.
092   * @param cuts the Alphabet cuts.
093   * @param nThreshold the normalization threshold value.
094   * 
095   * @return SAX representation of the time series.
096   */
097  public SAXRecords ts2saxByChunking(double[] ts, int paaSize, double[] cuts, double nThreshold) {
098
099    SAXRecords saxFrequencyData = new SAXRecords();
100
101    // Z normalize it
102    double[] normalizedTS = tsProcessor.znorm(ts, nThreshold);
103
104    // perform PAA conversion if needed
105    double[] paa = tsProcessor.paa(normalizedTS, paaSize);
106
107    // Convert the PAA to a string.
108    char[] currentString = tsProcessor.ts2String(paa, cuts);
109
110    // create the datastructure
111    for (int i = 0; i < currentString.length; i++) {
112      char c = currentString[i];
113      saxFrequencyData.add(String.valueOf(c).toCharArray(), i);
114    }
115
116    return saxFrequencyData;
117
118  }
119
120  /**
121   * Check for mindist.
122   * 
123   * @param a first string.
124   * @param b second string.
125   * @return true if mindist between strings is zero.
126   */
127  public boolean checkMinDistIsZero(char[] a, char[] b) {
128    for (int i = 0; i < a.length; i++) {
129      if (charDistance(a[i], b[i]) > 1) {
130        return false;
131      }
132    }
133    return true;
134  }
135
136  /**
137   * Convert the timeseries into SAX string representation.
138   * 
139   * @param ts the timeseries.
140   * @param paaSize the PAA size.
141   * @param cuts the alphabet cuts.
142   * @param nThreshold the normalization thresholds.
143   * 
144   * @return The SAX representation for timeseries.
145   */
146  public char[] ts2string(double[] ts, int paaSize, double[] cuts, double nThreshold) {
147
148    int tsLength = ts.length;
149    if (tsLength == paaSize) {
150      return tsProcessor.ts2String(tsProcessor.znorm(ts, nThreshold), cuts);
151    }
152    else {
153      // perform PAA conversion
154      double[] paa = tsProcessor.paa(ts, paaSize);
155      return tsProcessor.ts2String(paa, cuts);
156    }
157  }
158
159  /**
160   * Compute the distance between the two chars based on the ASCII symbol codes.
161   * 
162   * @param a The first char.
163   * @param b The second char.
164   * @return The distance.
165   */
166  public int charDistance(char a, char b) {
167    return Math.abs(Character.getNumericValue(a) - Character.getNumericValue(b));
168  }
169
170  /**
171   * Compute the distance between the two strings, this function use the numbers associated with
172   * ASCII codes, i.e. distance between a and b would be 1.
173   * 
174   * @param a The first string.
175   * @param b The second string.
176   * @return The pairwise distance.
177   * @throws SAXException if length are differ.
178   */
179  public int strDistance(char[] a, char[] b) throws SAXException {
180    if (a.length == b.length) {
181      int distance = 0;
182      for (int i = 0; i < a.length; i++) {
183        int tDist = Math.abs(Character.getNumericValue(a[i]) - Character.getNumericValue(b[i]));
184        if (tDist > 1) {
185          distance += tDist;
186        }
187      }
188      return distance;
189    }
190    else {
191      throw new SAXException("Unable to compute SAX distance, string lengths are not equal");
192    }
193  }
194
195  /**
196   * This function implements SAX MINDIST function which uses alphabet based distance matrix.
197   * 
198   * @param a The SAX string.
199   * @param b The SAX string.
200   * @param distanceMatrix The distance matrix to use.
201   * @return distance between strings.
202   * @throws SAXException If error occurs.
203   */
204  public double saxMinDist(char[] a, char[] b, double[][] distanceMatrix) throws SAXException {
205    if (a.length == b.length) {
206      double dist = 0.0D;
207      for (int i = 0; i < a.length; i++) {
208        if (Character.isLetter(a[i]) && Character.isLetter(b[i])) {
209          int numA = Character.getNumericValue(a[i]) - 10;
210          int numB = Character.getNumericValue(b[i]) - 10;
211          if (numA > 19 || numA < 0 || numB > 19 || numB < 0) {
212            throw new SAXException("The character index greater than 19 or less than 0!");
213          }
214          double localDist = distanceMatrix[numA][numB];
215          dist += localDist;
216        }
217        else {
218          throw new SAXException("Non-literal character found!");
219        }
220      }
221      return dist;
222    }
223    else {
224      throw new SAXException("Data arrays lengths are not equal!");
225    }
226  }

228  /**
229   * Generic method to convert the milliseconds into the elapsed time string.
230   * 
231   * @param start Start timestamp.
232   * @param finish End timestamp.
233   * @return String representation of the elapsed time.
234   */
235  public static String timeToString(long start, long finish) {
236
237    Duration duration = new Duration(finish - start); // in milliseconds
238    PeriodFormatter formatter = new PeriodFormatterBuilder().appendDays().appendSuffix("d")
239        .appendHours().appendSuffix("h").appendMinutes().appendSuffix("m").appendSeconds()
240        .appendSuffix("s").appendMillis().appendSuffix("ms").toFormatter();
241    String formatted = formatter.print(duration.toPeriod());
242
243    return formatted;
244
245  }
