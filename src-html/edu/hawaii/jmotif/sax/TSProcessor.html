044  /**
045   * Reads timeseries from a file. Assumes that file has a single double value on every line.
046   * Assigned timestamps are the line numbers.
047   * 
048   * @param filename The file to read from.
049   * @param columnIdx The column index.
050   * @param sizeLimit The number of lines to read, 0 == all.
051   * @return data.
052   * @throws NumberFormatException if error occurs.
053   * @throws IOException if error occurs.
054   * @throws TSException if error occurs.
055   */
056  public static double[] readFileColumn(String filename, int columnIdx, int sizeLimit)
057      throws NumberFormatException, IOException, SAXException {
058    BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(filename),
059        "UTF-8"));
060
061    ArrayList<Double> preRes = new ArrayList<Double>();
062    int lineCounter = 0;
063
064    String line = null;
065    while ((line = br.readLine()) != null) {
066      String[] split = line.split("\\s+");
067      if (split.length < columnIdx) {
068        String message = "Unable to read data from column " + columnIdx + " of file " + filename;
069        br.close();
070        throw new SAXException(message);
071      }
072      preRes.add(Double.valueOf(line));
073      lineCounter++;
074      if ((0 != sizeLimit) && (lineCounter >= sizeLimit)) {
075        break;
076      }
077    }
078    br.close();
079    double[] res = new double[preRes.size()];
080    for (int i = 0; i < preRes.size(); i++) {
081      res[i] = preRes.get(i);
082    }
083    return res;
084  }

086  /**
087   * Finds the maximal value in timeseries.
088   * 
089   * @param series The timeseries.
090   * @return The max value.
091   */
092  public double max(double[] series) {
093    double max = Double.MIN_VALUE;
094    for (int i = 0; i < series.length; i++) {
095      if (max < series[i]) {
096        max = series[i];
097      }
098    }
099    return max;
100  }

118  /**
119   * Computes the mean value of timeseries.
120   * 
121   * @param series The timeseries.
122   * @return The mean value.
123   */
124  public double mean(double[] series) {
125    double res = 0D;
126    int count = 0;
127    for (double tp : series) {
128      res += tp;
129      count += 1;
130
131    }
132    if (count > 0) {
133      return res / ((Integer) count).doubleValue();
134    }
135    return Double.NaN;
136  }

138  /**
139   * Compute the variance of timeseries.
140   * 
141   * @param series The timeseries.
142   * @return The variance.
143   */
144  public double var(double[] series) {
145    double res = 0D;
146    double mean = mean(series);
147    int count = 0;
148    for (double tp : series) {
149      res += (tp - mean) * (tp - mean);
150      count += 1;
151    }
152    if (count > 0) {
153      return res / ((Integer) (count - 1)).doubleValue();
154    }
155    return Double.NaN;
156  }

158  /**
159   * Speed-optimized implementation.
160   * 
161   * @param series The timeseries.
162   * @return the standard deviation.
163   */
164  public double stDev(double[] series) {
165    double num0 = 0D;
166    double sum = 0D;
167    int count = 0;
168    for (double tp : series) {
169      num0 = num0 + tp * tp;
170      sum = sum + tp;
171      count += 1;
172    }
173    double len = ((Integer) count).doubleValue();
174    return Math.sqrt((len * num0 - sum * sum) / (len * (len - 1)));
175  }

177  /**
178   * Computes the autocorrelation value of timeseries. according to algorithm in:
179   * http://www.itl.nist.gov/div898/handbook/eda/section3/eda35c.htm
180   * 
181   * @param series The timeseries.
182   * @param lag The lag
183   * @return The autocorrelation value.
184   */
185  public double autocorrelation(double series[], int lag) {
186    double ac = 0;
187
188    double avg = mean(series);
189    double numerator = 0;
190    for (int i = 0; i < series.length - lag; i++) {
191      if (Double.isNaN(series[i]) || Double.isInfinite(series[i])) {
192        continue;
193      }
194      numerator += (series[i] - avg) * (series[i + lag] - avg);
195    }
196    double denominator = 0;
197    for (int i = 0; i < series.length; i++) {
198      if (Double.isNaN(series[i]) || Double.isInfinite(series[i])) {
199        continue;
200      }
201      denominator += (series[i] - avg) * (series[i] - avg);
202    }
203    ac = numerator / denominator;
204    return ac;
205  }

207  /**
208   * Speed-optimized Z-Normalize routine, doesn't care about normalization threshold.
209   * 
210   * @param series The timeseries.
211   * @param normalizationThreshold
212   * @return Z-normalized time-series.
213   * @throws TSException if error occurs.
214   */
215  public double[] znorm(double[] series, double normalizationThreshold) {
216    double[] res = new double[series.length];
217    double mean = mean(series);
218    double sd = stDev(series);
219    if (sd < normalizationThreshold) {
220      return series.clone();
221    }
222    for (int i = 0; i < res.length; i++) {
223      res[i] = (series[i] - mean) / sd;
224    }
225    return res;
226  }

228  /**
229   * Approximate the timeseries using PAA. If the timeseries has some NaN's they are handled as
230   * follows: 1) if all values of the piece are NaNs - the piece is approximated as NaN, 2) if there
231   * are some (more or equal one) values happened to be in the piece - algorithm will handle it as
232   * usual - getting the mean.
233   * 
234   * @param ts The timeseries to approximate.
235   * @param paaSize The desired length of approximated timeseries.
236   * @return PAA-approximated timeseries.
237   */
238  public double[] paa(double[] ts, int paaSize) {
239    // fix the length
240    int len = ts.length;
241    // check for the trivial case
242    if (len == paaSize) {
243      return Arrays.copyOf(ts, ts.length);
244    }
245    else {
246      if (len % paaSize == 0) {
247        return colMeans(reshape(asMatrix(ts), len / paaSize, paaSize));
248      }
249      else {
250        double[] paa = new double[paaSize];
251        for (int i = 0; i < len * paaSize; i++) {
252          int idx = i / len; // the spot
253          int pos = i / paaSize; // the col spot
254          paa[idx] = paa[idx] + ts[pos];
255        }
256        for (int i = 0; i < paaSize; i++) {
257          paa[i] = paa[i] / (double) len;
258        }
259        return paa;
260      }
261    }
262
263  }

265  /**
266   * Converts the timeseries into string using given cuts intervals. Useful for not-normal
267   * distribution cuts.
268   * 
269   * @param vals The timeseries.
270   * @param cuts The cut intervals.
271   * @return The timeseries SAX representation.
272   */
273  public char[] ts2String(double[] vals, double[] cuts) {
274    char[] res = new char[vals.length];
275    for (int i = 0; i < vals.length; i++) {

282   * Get mapping of a number to char.
283   * 
284   * @param value the value to map.
285   * @param cuts the array of intervals.
286   * @return character corresponding to numeric value.
287   */
288  public char num2char(double value, double[] cuts) {
289    int count = 0;
290    while ((count < cuts.length) && (cuts[count] <= value)) {
291      count++;
292    }
293    return ALPHABET[count];
294  }

321  /**
322   * Extract subseries out of series.
323   * 
324   * @param series The series array.
325   * @param start Start position
326   * @param length Length of subseries to extract.
327   * @return The subseries.
328   * @throws IndexOutOfBoundsException If error occurs.
329   */
330  public double[] subseriesByCopy(double[] series, int start, int end)
331      throws IndexOutOfBoundsException {
332    if ((start > end) || (start < 0) || (end > series.length)) {
333      throw new IndexOutOfBoundsException("Unable to extract subseries, series length: "
334          + series.length + ", start: " + start + ", end: " + String.valueOf(end - start));
335    }
336    return Arrays.copyOfRange(series, start, end);
337  }

339  /**
340   * Implements Gaussian smoothing.
341   * 
342   * @param series Data to process.
343   * @param filterWidth The filter width.
344   * @return smoothed series.
345   * @throws SAXException if error occurs.
346   */
347  public double[] gaussFilter(double[] series, double filterWidth) throws SAXException {
348
349    double[] smoothedSignal = new double[series.length];
350    double sigma = filterWidth / 2D;
351    int maxShift = (int) Math.floor(4D * sigma); // Gaussian curve is reasonably > 0
352
353    if (maxShift < 1) {
354      throw new SAXException("NOT smoothing: filter width too small - " + filterWidth);
355    }
356    for (int i = 0; i < smoothedSignal.length; i++) {
357      smoothedSignal[i] = series[i];
358
359      if (maxShift < 1) {
360        continue;
361      }
362      for (int j = 1; j <= maxShift; j++) {
363
364        double gaussFilter = Math.exp(-(j * j) / (2. * sigma * sigma));
365        double leftAmpl, rightAmpl;
366
367        // go left
368        if ((i - j) >= 0) {
369          leftAmpl = series[i - j];
370        }
371        else {
372          leftAmpl = series[i];
373        }
374
375        // go right
376        if ((i + j) <= smoothedSignal.length - 1) {
377          rightAmpl = series[i + j];
378        }
379        else {
380          rightAmpl = series[i];
381        }
382
383        smoothedSignal[i] += gaussFilter * (leftAmpl + rightAmpl);
384
385      }
386
387      double normalizingCoef = Math.sqrt(2. * Math.PI) * sigma;
388      smoothedSignal[i] /= normalizingCoef;
389
390    }
391    return smoothedSignal;
392  }
393
394  public double gaussian(double x, double filterWidth) {
395    double sigma = filterWidth / 2.;
396    return Math.exp(-(x * x) / (2. * sigma * sigma));
397  }

399  public String seriesToString(double[] series, NumberFormat df) {
400    StringBuffer sb = new StringBuffer();
401    sb.append('[');
402    for (double d : series) {
403      sb.append(df.format(d)).append(',');
404    }
405    sb.delete(sb.length() - 2, sb.length() - 1).append("]");
406    return sb.toString();
407  }

409  /**
410   * Returns the m-by-n matrix B whose elements are taken column-wise from A. An error results if A
411   * does not have m*n elements.
412   * 
413   * @param a the source matrix.
414   * @param n number of rows in the new matrix.
415   * @param m number of columns in the new matrix.
416   * 
417   * @return reshaped matrix.
418   */
419  public double[][] reshape(double[][] a, int n, int m) {
420    int currentElement = 0;
421    int aRows = a.length;
422
423    double[][] res = new double[n][m];
424
425    for (int j = 0; j < m; j++) {
426      for (int i = 0; i < n; i++) {
427        res[i][j] = a[currentElement % aRows][currentElement / aRows];
428        currentElement++;
429      }
430    }
431    return res;
432  }

448  /**
449   * Computes column means for the matrix.
450   * 
451   * @param a the input matrix.
452   * @return result.
453   */
454  public double[] colMeans(double[][] a) {
455    double[] res = new double[a[0].length];
456    for (int j = 0; j < a[0].length; j++) {
457      double sum = 0;
458      int counter = 0;
459      for (int i = 0; i < a.length; i++) {
460        if (Double.isNaN(a[i][j]) || Double.isInfinite(a[i][j])) {
461          continue;
462        }
463        sum += a[i][j];
464        counter += 1;
465      }
466      if (counter == 0) {
467        res[j] = Double.NaN;
468      }
469      else {
470        res[j] = sum / ((Integer) counter).doubleValue();
471      }
472    }
473    return res;
474  }
