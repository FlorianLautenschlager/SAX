001package net.seninp.util;
002
003import java.io.PrintWriter;
004import java.io.StringWriter;
005
006/**
007 * Provides a simple solution to the common problem of obtaining a String containing the stack trace
008 * produced by an exception.
009 * <p>
010 * Call StackTrace.toString(e) to get the string corresponding to the Exception e.
011 * 
012 * @author Philip Johnson, Takuya Yamashita
013 */
014public final class StackTrace {
015
016  /** Disable public constructor. */
017  private StackTrace() {
018    // do nothing
019  }
020
021  /**
022   * Converts the Throwable.getStackTrace to a String representation for logging.
023   * 
024   * @param throwable The Throwable exception.
025   * @return A String containing the StackTrace.
026   */
027  public static String toString(Throwable throwable) {
028    StringWriter stringWriter = new StringWriter();
029    throwable.printStackTrace(new PrintWriter(stringWriter));
030    return stringWriter.toString();
031  }
032}
