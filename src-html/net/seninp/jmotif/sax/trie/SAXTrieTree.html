001package net.seninp.jmotif.sax.trie;
002
003import java.util.List;
004
005/**
006 * Implements the magic trie structure.
007 * 
008 * @author Pavel Senin.
009 * 
010 */
011public class SAXTrieTree {
012
013  /**
014   * Constants for the alphabet. Where it starts and ends. English.
015   */
016  private static final Integer ALPHABET_CODE_START = 97;
017  private static final Integer ALPHABET_CODE_END = 122;
018
019  /** The root of the tree. */
020  private TrieInnerNode root;
021
022  /** The alphabet size used for the building - the maximal word length. */
023  private Integer alphabetSize;
024
025  /**
026   * Constructor.
027   * 
028   * @param alphabetSize The alphabet size.
029   * @throws TrieException If wrong parameters specified.
030   */
031  public SAXTrieTree(Integer alphabetSize) throws TrieException {
032    if ((null != alphabetSize) && (alphabetSize > 0)
033        && (alphabetSize < (ALPHABET_CODE_END - ALPHABET_CODE_START))) {
034      this.alphabetSize = alphabetSize;
035      root = new TrieInnerNode("root");
036      buildTrie(root, alphabetSize, alphabetSize);
037    }
038    else {
039      throw new TrieException("Unable to create trie structure for the alphabet size of "
040          + alphabetSize);
041    }
042  }
043
044  /**
045   * Traverse the tree following the string and get occurrences list.
046   * 
047   * @param string The query string.
048   * @return The occurrences array.
049   * @throws TrieException If error occurs - wrong string length provided etc.
050   */
051  public List<Integer> getOccurrences(String string) throws TrieException {
052
053    // sanity check
054    //
055    if (this.root.getDescendants().size() > 0 && string.length() > 0
056        && this.alphabetSize == string.length() && !containsWrongSymbols(string)) {
057
058      // if string length is 2 - just pull an answer
059      //
060      if (2 == string.length()) {
061        return ((TrieLeafNode) ((TrieInnerNode) this.root.getDescendant(string.substring(0, 1)))
062            .getDescendant(string.substring(1))).getOccurences();
063      }
064
065      // if length greater than 2 - call the method recursively
066      //
067      String keyPrefix = string.substring(0, 1);
068      String suffix = string.substring(1);
069      return getSubstringOccurrence(this.root.getDescendant(keyPrefix), suffix);
070
071    }
072    else {
073      throw new TrieException("Unable to get occurences for the string \"" + string
074          + "\" the tree depth (alphabet size) is " + this.alphabetSize);
075    }
076  }
077
078  /**
079   * Add the occurrence into the table.
080   * 
081   * @param str The string.
082   * @param idx The occurrence index.
083   * @return the full list of occurrences.
084   * @throws TrieException if goes wrong.
085   */
086  public List<Integer> addOccurence(String str, int idx) throws TrieException {
087
088    // sanity check
089    //
090    if ((null == str) || (str.length() != this.alphabetSize) || (containsWrongSymbols(str))) {
091      throw new TrieException("Cannot populate occurrence of \"" + str + "\"into the trie of size "
092          + this.alphabetSize);
093    }
094
095    // first wee need to see if the string is going to be accepted
096    // so we do traverse till the last symbol of the string
097    //
098    String prefix = str.substring(0, 1);
099    String rest = str.substring(1);
100
101    // pick the first internal node corresponding to the first character
102    // WE HAVE TO DO THIS SINCE ROOT NODE IS DIFFERENT A BIT
103    //
104    TrieAbstractNode cNode = this.root;
105
106    // traverse the tree in the loop
107    //
108    while ((rest.length() > 0) && (cNode = ((TrieInnerNode) cNode).getDescendant(prefix)) != null) {
109      // if string is larger than 1 symbol - go deeper
110      prefix = rest.substring(0, 1);
111      rest = rest.substring(1);
112    }
113
114    // check that traversal finished properly
115    //
116    if (null == cNode) {
117      throw new TrieException("Internal error: having a null node where it shouldn't be.");
118    }
119
120    // here last character left - get the array of occurrences and add the new one
121    //
122    TrieLeafNode leaf = (TrieLeafNode) ((TrieInnerNode) cNode).getDescendant(prefix);
123    leaf.addOccurrence(idx);
124    return leaf.getOccurences();
125  }
126
127  /**
128   * Build the actual trie.
129   * 
130   * @param root The root of the current tree.
131   * @param alphabetSize The alphabet size.
132   * @param depth2Go The depth of the tree left.
133   */
134  private void buildTrie(TrieAbstractNode root, Integer alphabetSize, Integer depth2Go) {
135    // if depth allows
136    //
137    if (depth2Go > 1) {
138      //
139      // internal trie nodes
140      // create nodes for the alphabet size and recursively call further
141      for (int i = 0; i < alphabetSize; i++) {
142        // this char is the one from aStart + a value between 0 and the alphabet size
143        char curChar = (char) (i + ALPHABET_CODE_START);
144        // instantiate and add the node to the descendants list
145        TrieInnerNode node = new TrieInnerNode(String.valueOf(curChar));
146        ((TrieInnerNode) root).addNext(node);
147        // recursively call the build
148        buildTrie(node, alphabetSize, depth2Go - 1);
149      }
150    }
151    else {
152      //
153      // depth2Go == 1
154      //
155      // nodes after this one must be leafs
156      for (int i = 0; i < alphabetSize; i++) {
157        char curChar = (char) (i + ALPHABET_CODE_START);
158        // so we put them at place
159        TrieLeafNode node = new TrieLeafNode(String.valueOf(curChar));
160        ((TrieInnerNode) root).addNext(node);
161      }
162    }
163  }
164
165  /**
166   * Internal method used for the tree traversal.
167   * 
168   * @param root The current node, IT IS NOT ROOT NODE, we just call it root here due to context.
169   * @param str The string to use for traversal from this node.
170   * @return list of found occurrences.
171   * @throws TrieException If error occurs.
172   */
173  private List<Integer> getSubstringOccurrence(TrieAbstractNode root, String str)
174      throws TrieException {
175    if (TrieNodeType.INNER.equals(root.getType()) && str.length() > 1) {
176      // normal case - digging deeper
177      String keyPrefix = str.substring(0, 1);
178      String suffix = str.substring(1);
179      return getSubstringOccurrence(((TrieInnerNode) root).getDescendant(keyPrefix), suffix);
180    }
181    else if (TrieNodeType.INNER.equals(root.getType()) && str.length() == 1) {
182      // string length is 1
183      return ((TrieLeafNode) ((TrieInnerNode) root).getDescendant(str)).getOccurences();
184    }
185    else {
186      throw new TrieException("Ubnormal condition passed into the method. Unable to proceed.");
187    }
188  }
189
190  /**
191   * The string validator - check if string has only proper letters.
192   * 
193   * @param str The string to check.
194   * @return True if string contains only acceptable (valid) symbols.
195   */
196  private boolean containsWrongSymbols(String str) {
197    int maxSymbol = ALPHABET_CODE_START + alphabetSize;
198    for (int i = 0; i < str.length(); i++) {
199      if (((int) str.charAt(i)) >= maxSymbol) {
200        return true;
201      }
202    }
203    return false;
204  }
205}
