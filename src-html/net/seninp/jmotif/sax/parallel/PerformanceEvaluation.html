001package net.seninp.jmotif.sax.parallel;
002
003import net.seninp.jmotif.sax.NumerosityReductionStrategy;
004import net.seninp.jmotif.sax.SAXProcessor;
005import net.seninp.jmotif.sax.TSProcessor;
006import net.seninp.jmotif.sax.alphabet.NormalAlphabet;
007import net.seninp.jmotif.sax.datastructures.SAXRecords;
008
009/**
010 * This runs the performance evaluation code - we are looking on the speedup.
011 * 
012 * @author psenin
013 * 
014 */
015public final class PerformanceEvaluation {
016
017  private static final Integer NRUNS = 20;
018
019  private static final Integer MIN_CPUS = 2;
020
021  private static final Integer MAX_CPUS = 16;
022
023  private static final double N_THRESHOLD = 0.001d;
024
025  /**
026   * Runs the evaluation.
027   * 
028   * @param args some accepted, see the code.
029   * @throws Exception thrown if an error occured.
030   */
031  public static void main(String[] args) throws Exception {
032
033    NormalAlphabet na = new NormalAlphabet();
034    SAXProcessor sp = new SAXProcessor();
035
036    String dataFileName = args[0];
037    Integer slidingWindowSize = Integer.valueOf(args[1]);
038    Integer paaSize = Integer.valueOf(args[2]);
039    Integer alphabetSize = Integer.valueOf(args[3]);
040    NumerosityReductionStrategy nrStrategy = NumerosityReductionStrategy.valueOf(args[4]
041        .toUpperCase());
042
043    double[] ts = TSProcessor.readFileColumn(dataFileName, 0, 0);
044    System.out.println("data file: " + dataFileName);
045    System.out.println("data size: " + ts.length);
046    System.out.println("SAX parameters: sliding window size " + slidingWindowSize + ", PAA size "
047        + paaSize + ", alphabet size " + alphabetSize + ", NR strategy " + nrStrategy.toString());
048
049    System.out.println("Performing " + NRUNS
050        + " SAX conversion runs for each algorithm implementation ... ");
051
052    // conventional
053    //
054    long tstamp1 = System.currentTimeMillis();
055    for (int i = 0; i < NRUNS; i++) {
056      @SuppressWarnings("unused")
057      SAXRecords sequentialRes = sp.ts2saxViaWindow(ts, slidingWindowSize, paaSize,
058          na.getCuts(alphabetSize), NumerosityReductionStrategy.EXACT, N_THRESHOLD);
059    }
060    long tstamp2 = System.currentTimeMillis();
061    System.out.println("single thread conversion: " + String.valueOf(tstamp2 - tstamp1) + ", "
062        + SAXProcessor.timeToString(tstamp1, tstamp2));
063
064    // parallel
065    for (int threadsNum = MIN_CPUS; threadsNum < MAX_CPUS; threadsNum++) {
066      tstamp1 = System.currentTimeMillis();
067      for (int i = 0; i < NRUNS; i++) {
068        ParallelSAXImplementation ps = new ParallelSAXImplementation();
069        @SuppressWarnings("unused")
070        SAXRecords parallelRes = ps.process(ts, threadsNum, slidingWindowSize, paaSize,
071            alphabetSize, NumerosityReductionStrategy.EXACT, N_THRESHOLD);
072      }
073      tstamp2 = System.currentTimeMillis();
074      System.out.println("parallel conversion using " + threadsNum + " threads: "
075          + String.valueOf(tstamp2 - tstamp1) + ", " + SAXProcessor.timeToString(tstamp1, tstamp2));
076    }
077  }
078
079}
