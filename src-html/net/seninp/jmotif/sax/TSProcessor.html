052  /**
053   * Reads timeseries from a file. Assumes that file has a single double value on every line.
054   * Assigned timestamps are the line numbers.
055   * 
056   * @param filename The file to read from.
057   * @param columnIdx The column index.
058   * @param sizeLimit The number of lines to read, 0 == all.
059   * @return data.
060   * @throws NumberFormatException if error occurs.
061   * @throws IOException if error occurs.
062   * @throws SAXException if error occurs.
063   */
064  public static double[] readFileColumn(String filename, int columnIdx, int sizeLimit)
065      throws NumberFormatException, IOException, SAXException {
066    BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(filename),
067        "UTF-8"));
068
069    ArrayList<Double> preRes = new ArrayList<Double>();
070    int lineCounter = 0;
071
072    String line = null;
073    while ((line = br.readLine()) != null) {
074      String[] split = line.split("\\s+");
075      if (split.length < columnIdx) {
076        String message = "Unable to read data from column " + columnIdx + " of file " + filename;
077        br.close();
078        throw new SAXException(message);
079      }
080      preRes.add(Double.valueOf(line));
081      lineCounter++;
082      if ((0 != sizeLimit) && (lineCounter >= sizeLimit)) {
083        break;
084      }
085    }
086    br.close();
087    double[] res = new double[preRes.size()];
088    for (int i = 0; i < preRes.size(); i++) {
089      res[i] = preRes.get(i);
090    }
091    return res;
092  }

094  /**
095   * Finds the maximal value in timeseries.
096   * 
097   * @param series The timeseries.
098   * @return The max value.
099   */
100  public double max(double[] series) {
101    double max = Double.MIN_VALUE;
102    for (int i = 0; i < series.length; i++) {
103      if (max < series[i]) {
104        max = series[i];
105      }
106    }
107    return max;
108  }

126  /**
127   * Computes the mean value of timeseries.
128   * 
129   * @param series The timeseries.
130   * @return The mean value.
131   */
132  public double mean(double[] series) {
133    double res = 0D;
134    int count = 0;
135    for (double tp : series) {
136      res += tp;
137      count += 1;
138
139    }
140    if (count > 0) {
141      return res / ((Integer) count).doubleValue();
142    }
143    return Double.NaN;
144  }

146  /**
147   * Compute the variance of timeseries.
148   * 
149   * @param series The timeseries.
150   * @return The variance.
151   */
152  public double var(double[] series) {
153    double res = 0D;
154    double mean = mean(series);
155    int count = 0;
156    for (double tp : series) {
157      res += (tp - mean) * (tp - mean);
158      count += 1;
159    }
160    if (count > 0) {
161      return res / ((Integer) (count - 1)).doubleValue();
162    }
163    return Double.NaN;
164  }

166  /**
167   * Speed-optimized implementation.
168   * 
169   * @param series The timeseries.
170   * @return the standard deviation.
171   */
172  public double stDev(double[] series) {
173    double num0 = 0D;
174    double sum = 0D;
175    int count = 0;
176    for (double tp : series) {
177      num0 = num0 + tp * tp;
178      sum = sum + tp;
179      count += 1;
180    }
181    double len = ((Integer) count).doubleValue();
182    return Math.sqrt((len * num0 - sum * sum) / (len * (len - 1)));
183  }

185  /**
186   * Speed-optimized Z-Normalize routine, doesn't care about normalization threshold.
187   * 
188   * @param series The timeseries.
189   * @param normalizationThreshold the zNormalization threshold value.
190   * @return Z-normalized time-series.
191   */
192  public double[] znorm(double[] series, double normalizationThreshold) {
193    double[] res = new double[series.length];
194    double mean = mean(series);
195    double sd = stDev(series);
196    if (sd < normalizationThreshold) {
197      return series.clone();
198    }
199    for (int i = 0; i < res.length; i++) {
200      res[i] = (series[i] - mean) / sd;
201    }
202    return res;
203  }

205  /**
206   * Approximate the timeseries using PAA. If the timeseries has some NaN's they are handled as
207   * follows: 1) if all values of the piece are NaNs - the piece is approximated as NaN, 2) if there
208   * are some (more or equal one) values happened to be in the piece - algorithm will handle it as
209   * usual - getting the mean.
210   * 
211   * @param ts The timeseries to approximate.
212   * @param paaSize The desired length of approximated timeseries.
213   * @return PAA-approximated timeseries.
214   */
215  public double[] paa(double[] ts, int paaSize) {
216    // fix the length
217    int len = ts.length;
218    // check for the trivial case
219    if (len == paaSize) {
220      return Arrays.copyOf(ts, ts.length);
221    }
222    else {
223      if (len % paaSize == 0) {
224        return colMeans(reshape(asMatrix(ts), len / paaSize, paaSize));
225      }
226      else {
227        double[] paa = new double[paaSize];
228        for (int i = 0; i < len * paaSize; i++) {
229          int idx = i / len; // the spot
230          int pos = i / paaSize; // the col spot
231          paa[idx] = paa[idx] + ts[pos];
232        }
233        for (int i = 0; i < paaSize; i++) {
234          paa[i] = paa[i] / (double) len;
235        }
236        return paa;
237      }
238    }

242  /**
243   * Converts the timeseries into string using given cuts intervals. Useful for not-normal
244   * distribution cuts.
245   * 
246   * @param vals The timeseries.
247   * @param cuts The cut intervals.
248   * @return The timeseries SAX representation.
249   */
250  public char[] ts2String(double[] vals, double[] cuts) {
251    char[] res = new char[vals.length];
252    for (int i = 0; i < vals.length; i++) {
253      res[i] = num2char(vals[i], cuts);
254    }
255    return res;
256  }

258  /**
259   * Convert the timeseries into the index using SAX cuts.
260   * 
261   * @param series The timeseries to convert.
262   * @param alphabet The alphabet to use.
263   * @param alphabetSize The alphabet size in use.
264   * @return SAX representation of timeseries.
265   * @throws Exception if error occurs.
266   */
267  public int[] ts2Index(double[] series, Alphabet alphabet, int alphabetSize) throws Exception {
268    double[] cuts = alphabet.getCuts(alphabetSize);
269    int[] res = new int[series.length];
270    for (int i = 0; i < series.length; i++) {
271      res[i] = num2index(series[i], cuts);
272    }
273    return res;
274  }

276  /**
277   * Get mapping of a number to char.
278   * 
279   * @param value the value to map.
280   * @param cuts the array of intervals.
281   * @return character corresponding to numeric value.
282   */
283  public char num2char(double value, double[] cuts) {
284    int count = 0;
285    while ((count < cuts.length) && (cuts[count] <= value)) {
286      count++;
287    }
288    return ALPHABET[count];
289  }

316  /**
317   * Extract subseries out of series.
318   * 
319   * @param series The series array.
320   * @param start the fragment start.
321   * @param end the fragment end.
322   * @return The subseries.
323   * @throws IndexOutOfBoundsException If error occurs.
324   */
325  public double[] subseriesByCopy(double[] series, int start, int end)
326      throws IndexOutOfBoundsException {
327    if ((start > end) || (start < 0) || (end > series.length)) {
328      throw new IndexOutOfBoundsException("Unable to extract subseries, series length: "
329          + series.length + ", start: " + start + ", end: " + String.valueOf(end - start));
330    }
331    return Arrays.copyOfRange(series, start, end);
332  }

335   * Prettyfies the timeseries for screen output.
336   * 
337   * @param series the data.
338   * @param df the number format to use.
339   * 
340   * @return The timeseries formatted for screen output.
341   */
342  public String seriesToString(double[] series, NumberFormat df) {
343    StringBuffer sb = new StringBuffer();
344    sb.append('[');
345    for (double d : series) {
346      sb.append(df.format(d)).append(',');
347    }
348    sb.delete(sb.length() - 2, sb.length() - 1).append("]");
349    return sb.toString();
350  }

352  /**
353   * Returns the m-by-n matrix B whose elements are taken column-wise from A. An error results if A
354   * does not have m*n elements.
355   * 
356   * @param a the source matrix.
357   * @param n number of rows in the new matrix.
358   * @param m number of columns in the new matrix.
359   * 
360   * @return reshaped matrix.
361   */
362  public double[][] reshape(double[][] a, int n, int m) {
363    int currentElement = 0;
364    int aRows = a.length;
365
366    double[][] res = new double[n][m];
367
368    for (int j = 0; j < m; j++) {
369      for (int i = 0; i < n; i++) {
370        res[i][j] = a[currentElement % aRows][currentElement / aRows];
371        currentElement++;
372      }
373    }
374    return res;
375  }

391  /**
392   * Computes column means for the matrix.
393   * 
394   * @param a the input matrix.
395   * @return result.
396   */
397  public double[] colMeans(double[][] a) {
398    double[] res = new double[a[0].length];
399    for (int j = 0; j < a[0].length; j++) {
400      double sum = 0;
401      int counter = 0;
402      for (int i = 0; i < a.length; i++) {
403        if (Double.isNaN(a[i][j]) || Double.isInfinite(a[i][j])) {
404          continue;
405        }
406        sum += a[i][j];
407        counter += 1;
408      }
409      if (counter == 0) {
410        res[j] = Double.NaN;
411      }
412      else {
413        res[j] = sum / ((Integer) counter).doubleValue();
414      }
415    }
416    return res;
417  }

419  /**
420   * Read at least N elements from the file.
421   * 
422   * @param dataFileName the file name.
423   * @param loadLimit the load limit.
424   * @return the read data or empty array if nothing to load.
425   * @throws SAXException if error occurs.
426   */
427  public double[] readTS(String dataFileName, int loadLimit) throws SAXException {
428
429    // make sure the path exists
430    Path path = Paths.get(dataFileName);
431    if (!(Files.exists(path))) {
432      throw new SAXException("unable to load data - data source not found.");
433    }
434
435    // read the input
436    //
437    // init the data araay
438    ArrayList<Double> data = new ArrayList<Double>();
439
440    // lets go
441    try {
442
443      // open the reader
444      BufferedReader reader = Files.newBufferedReader(path, DEFAULT_CHARSET);
445
446      // read by the line in the loop from reader
447      String line = null;
448      long lineCounter = 0;
449      while ((line = reader.readLine()) != null) {
450        String[] lineSplit = line.trim().split("\\s+");
451        // we read only first column
452        // for (int i = 0; i < lineSplit.length; i++) {
453        double value = new BigDecimal(lineSplit[0]).doubleValue();
454        data.add(value);
455        // }
456        lineCounter++;
457        // break the load if needed
458        if ((loadLimit > 0) && (lineCounter > loadLimit)) {
459          break;
460        }
461      }
462      reader.close();
463    }
464    catch (Exception e) {
465      System.err.println(StackTrace.toString(e));
466    }
467    finally {
468      assert true;
469    }
470
471    // convert to simple doubles array and clean the variable
472    if (!(data.isEmpty())) {
473      double[] ts = new double[data.size()];
474      for (int i = 0; i < data.size(); i++) {
475        ts[i] = data.get(i);
476      }
477      return ts;
478    }
479    return new double[0];
480  }
